Where means 'h' for hexidecimal number

Standard Numeric Datatypes
------------------
BOOLEAN = 0001h
INTEGER8 = 0002h
INTEGER16 = 0003h
INTEGER24 = 0010h // not supported
INTEGER32 = 0004h
INTEGER40 = 0012h // not supported
INTEGER48 = 0013h // not supported
INTEGER56 = 0014h // not supported
INTEGER64 = 0015h
UNSIGNED8 = 0005h
UNSIGNED16 = 0006h
UNSIGNED24 = 0016h // not supported
UNSIGNED32 = 0007h
UNSIGNED40 = 0018h // not supported
UNSIGNED48 = 0019h // not supported
UNSIGNED56 = 001Ah // not supported
UNSIGNED64 = 001Bh
REAL32 = 0008h
REAL64 = 0011h
